        /* We have to change the heights on both the iframe and its containing div otherwise there's overflow clipping */
    
        function expandElement( $el, key ) {
            /* since multiple elements need to be expanded we create different keys for their respective timeouts so we can
                clear those timeouts if necessary */
    
            timeouts[key] = setTimeout(function(){
                $el.height(open_height);
            }, delay)
        }
    
        function collapseElement( el, key ) {
            clearTimeout(timeouts[key]);
            $(el).height(default_height);
        }
    
        if ( notTouch ){
            $container.on('mouseover', 'iframe', function(){
                var $el = $(this);
                /* this has to be in the on handler instead of outside because the iframe is added asynchronously */
                var w = $el.width();
                if ( w <=300 || w >= 940  ){
                    expandElement($el, 'iframe');                        
                }
            });
            $container.on('mouseout', 'iframe', function(){
                /* there's no check on the width in case user resizes the page (and therefore iframe) down */
                collapseElement(this, 'iframe');                   
            });
        }
    
    }

})(jQuery);
