			// This script sets the top nav to have equal spacing between each item

			var topNavTotalWidth = 792;
			var topNavLinksWidth = 0;				
			var $topNavLinks = $('#topNav td a');		
			var topNavNumLinks = $topNavLinks.length;
			
			// add up the total calculated width occupied by links
			$topNavLinks.each(function(i){		
			
				// set the link to no wrap and display block for proper width calc
				$(this).css({
					'text-align': 'left', 
					'display':'block', 
					'float':'left',
					'white-space':'nowrap'
				});
				
				topNavLinksWidth += $(this).width();

			// calculate the amount of spacing that should be added to the right of each link except the last
			var topNavSpacing = (topNavTotalWidth - topNavLinksWidth) / (topNavNumLinks - 1);
			
			$('#topNav td').each(function(i) {			
				
				// get the link width
				var linkWidth = $(this).find('a').width();
				
				// add the calculated spacing to the cell			
				var newCellWidth = linkWidth + topNavSpacing;
				
				// the last cell should not have spacing added and should get the same width as the link
				if(i == topNavNumLinks-1) {
					newCellWidth = linkWidth;
				}
				
				// set the new width of the cell and change the alignment to left
				$(this).css('text-align', 'left').width(newCellWidth);
			});
