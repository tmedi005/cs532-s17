                <div class="widget widget-header-text var-feel-vintage-text-standard" id="w-2187507863"><div class="content colored-background" data-clicktarget="#">    <h1 class="theme-title" data-target="title">        <span class="autofit"  style="font-size: 119%"  data-autofit-done="true" data-priority="lines" data-lines="2" data-max="+=20" >Wading in the Middle</span>    </h1>        <div class="subtitle-container">        <h3 class="theme-subtitle colored-background">            <span class="autofit"  style="font-size: 99%"  data-autofit-done="true" data-lines="1" data-priority="lines" data-target="subtitle" >The Upcoming Weeks in C7: February 21- March 3!</span>        </h3>    </div>    </div></div><div class="widget widget-text-paragraph var-paragraph-standard" id="w-0275533945"><div class="pagegrid">    <div class="row">        <div class="six columns alpha omega">            <div class="content">                                <h2 class="theme-p-title theme-color3" data-target="title" data-clicktarget="#">                Humanities                </h2>                                <div class="theme-p theme-color-4" data-target="content">                    <p><strong>What</strong>: We are learning about the types of governments while looking at the social and economic changes in the 20th and 21st century.</p><p><br/></p><p><strong>How:</strong> We are evaluating the driving question &quot;How do we become agents of change?&quot; Students will pick a country to research that has had a government overthrown in the last 117 years. They will learn about the previous type of government, the people who brought about change, and the government that was put in place. This should carry us through spring break.</p><p><br/></p><p><strong>Why</strong>: It is important for students to learn about all the differences in the world.         function TimeSpentTracker() {

            var PING_URL = '/app/reporting/ping/58a3231f8b1188019840541c';

            var SECOND = 1000;
            var MINUTE = SECOND * 60;
            var HOUR = MINUTE * 60;

            var INITIAL_TIME_INTERVAL = 5 * SECOND;
            var MAX_TRACKING_TIME_LIMIT = 2 * HOUR;

            this.pingTimerId;
            this.pingTimeInterval = INITIAL_TIME_INTERVAL;

            this.start = function () {
                this.ping();
                this.runPingLoop()
            };

            this.stop = function () {
                window.clearInterval(this.pingTimerId);
            };

            this.runPingLoop = function () {
                var self = this;
                this.pingTimerId = window.setTimeout(function () {
                    self.ping()
                }, this.pingTimeInterval);
            };

            this.ping = function () {
                this.pingServer();
                if (this.timeLimitReached()) {
                    this.stop();
                } else {
                    this.incrementPingTimeInterval();
                    this.runPingLoop();
                }
            };

            this.timeLimitReached = function () {
                return this.pingTimeInterval > MAX_TRACKING_TIME_LIMIT;
            };

            /**
             * 2X time : from 5 sec to 10 min (5, 10, 20, 40, 80, 160, ...)
             * +10 min : from 10 min to 1 hour
             * +15 min : from 1 hour to MAX_TRACKING_TIME_LIMIT
             */
            this.incrementPingTimeInterval = function () {
                if (this.pingTimeInterval >= 10 * MINUTE) {
                    this.pingTimeInterval += 10 * MINUTE;
                } else if (this.pingTimeInterval >= HOUR) {
                    this.pingTimeInterval += 15 * MINUTE;
                } else {
                    this.pingTimeInterval *= 2;
                }
            };

            this.pingServer = function () {
                var beaconPixel = new Image();
                // We use a timestamp to invalidate image cache
                var timestamp = new Date().getTime();
                // Setting the Image.src hits the server
                beaconPixel.src = PING_URL + '?timestamp=' + timestamp;
            };
        }

        try {

            window.__smoreTimeTracker.start();
        } catch (e) {
            if (window.addEventListener) {
                window.addEventListener('time-spent-tracking:stop', function () {
                    window.__smoreTimeTracker.stop();
                });
            }
        }
